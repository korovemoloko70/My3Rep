   return [int(element) for element in bin(value)[2:].zfill(8)]

while True:
   if GPIO.input (up) :
       num +=1
       if num>255:
           num = 0
       print (num, dec2bin(num))

   if GPIO. input (down):
       num -=1
       if num<0:
           num = 0
       print (num, dec2bin(num))
       time.sleep(sleep_time)
   GPIO.output(leds, dec2bin(num))

get-led/jumping-led.py

import RPi.GPIO as GPIO
import time
GPIO.setmode(GPIO.BCM)

leds = [24, 22, 23, 27, 17, 25, 12, 16]
GPIO.setup(leds, GPIO.OUT)
GPIO.setup(leds, 0)
light_time=0.2

while True:
   for led in leds:
       GPIO.output(led, 1)
       time.sleep(light_time)
       GPIO.output(led, 0)
   for led in reversed(leds):
      GPIO.output(led, 1)
      time.sleep(light_time)
      GPIO.output(led,0)

import RPi.GPIO as GPIO
import time
GPIO.setmode(GPIO.BCM)
led = 26
GPIO.setup(led, GPIO.OUT)
pwm = GPIO.PWM (led, 200)
duty = 0.0
pwm.start(duty)
while True:
   pwm.ChangeDutyCycle(duty)
   time.sleep(0.05)
   duty +=1.0
   if duty >100.0:
       duty = 0.9
